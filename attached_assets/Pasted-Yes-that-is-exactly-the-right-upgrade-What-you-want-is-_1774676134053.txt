## Add a converter function

In your repo, the cleanest place is near the pipeline/session runner side.

Create something like:

```python
def decision_to_chart_event(decision) -> dict:
    return {
        "ts": decision.bar_ts.isoformat(),
        "instrument": decision.instrument,
        "setup_type": decision.setup_type,
        "direction": decision.direction,
        "decision_type": decision.decision_type.value,
        "scores": decision.scores.to_dict() if decision.scores else {},
        "entry_price": decision.entry_price,
        "sl_price": decision.sl_price,
        "tp_price": decision.tp_price,
        "labels": _extract_labels(decision),
        "reason": decision.error or "",
    }
```

And:

```python
def _extract_labels(decision) -> list[str]:
    labels = []
    if decision.claude_verdict:
        if not decision.claude_verdict.get("trade", True):
            labels.append("claude_veto")
    if decision.risk_decision:
        if decision.risk_decision.get("approved"):
            labels.append("risk_ok")
        elif decision.risk_decision.get("reason"):
            labels.append(decision.risk_decision["reason"])
    return labels
```
